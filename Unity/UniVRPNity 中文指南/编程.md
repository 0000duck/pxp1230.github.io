# AnalogRemoteMB：轴设备
```
// 鼠标轴范围：
// 0 ------> 1 
// |         x
// |
// V
// 1  y

public AnalogRemoteMB Analog;

Analog.AnalogChanged += new AnalogRemoteMB.AnalogChangeEventHandler(AnalogChanged);

public virtual void AnalogChanged(UniVRPNity.AnalogEvent e)
{
	//鼠标 or ART Flystick
	float x = (float)e.Channel((int)Mouse.Analog.X);
	float y = (float)e.Channel((int)Mouse.Analog.Y);
	
	//Wiimote
	float battery = (float)e.Channel((int)Wiimote.Analog.Battery);//电池电量
	float roll = (float)e.Channel((int)Wiimote.Analog.Roll);
	float pitch = (float)e.Channel((int)Wiimote.Analog.Pitch);
	float yaw = (float)e.Channel((int)Wiimote.Analog.Yaw);
	float roll2 = (float)e.Channel((int)Wiimote.Analog.NunchuckRoll);
	float pitch2 = (float)e.Channel((int)Wiimote.Analog.NunchuckPitch);
	float yaw2 = (float)e.Channel((int)Wiimote.Analog.NunchuckYaw);
}

//ART Flystick
float fsX = (float)Analog.Channel((int)Wiimote.Analog.FirstSensorSpotX);
float fsY = (float)Analog.Channel((int)Wiimote.Analog.FirstSensorSpotY);
float ssX = (float)Analog.Channel((int)Wiimote.Analog.SecondSensorSpotX);
float ssY = (float)Analog.Channel((int)Wiimote.Analog.SecondSensorSpotY);
```

# Y轴与Z轴对调
```
//右手坐标系→Unity的左手坐标系
Vector3 v3 = tracker.Position(1);
transform.position = new Vector3(v3.x, v3.z, v3.y);
Quaternion q4 = tracker.Orientation(1);
transform.rotation = new Quaternion(q4.x, q4.z, q4.y, -q4.w);
```
